  6. void cocktailSort(int arr[], int arrSize) {
  7. 	int firstIndex = 0; //индекс первого элемента рабочей зоны массива
  8. 	int lastIndex = arrSize-1; //индекс последнего элемента рабочей зоны массива
  9.  
  10. 	while (firstIndex < lastIndex) {
  11. 		/*проход слева направо*/
  12. 		for (int i = firstIndex; i < lastIndex; i++) {
  13. 			if (arr[i] > arr[i + 1]) {
  14. 				/*Обмен элементов*/
  15. 				int tempElement = arr[i];
  16. 				arr[i] = arr[i + 1];
  17. 				arr[i + 1] = tempElement;
  18. 			}
  19. 		}
  20. 		/*проход справо налево*/
  21. 		for (int i = lastIndex; i > firstIndex; i--) {
  22. 			if (arr[i] < arr[i - 1]) {
  23. 				/*Обмен элементов*/
  24. 				int tempElement = arr[i];
  25. 				arr[i] = arr[i - 1];
  26. 				arr[i - 1] = tempElement;
  27. 			}
  28. 		}
  29. 		firstIndex++;
  30. 		lastIndex--;
  31. 	}
  32. }
