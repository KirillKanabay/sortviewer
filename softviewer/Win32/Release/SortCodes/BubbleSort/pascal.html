1. program BubbleSortAlgorithm;
  2.  
  3. const
  4.   ArrSize = 10;
  5.  
  6. type
  7.   TArray = array [1..ArrSize] of integer;
  8.  
  9. var
  10.   Arr: TArray;
  11.  
  12. procedure BubbleSort(var arr: TArray);
  13. var
  14.   TempElement: integer;
  15. begin
  16.   for var i := 1 to ArrSize - 1 do
  17.     for var j := 1 to ArrSize - i do
  18.     begin
  19.       {проход слева направо}
  20.       if Arr[j] > Arr[j + 1] then begin
  21.         {обмен элементов}
  22.         TempElement := Arr[j];
  23.         Arr[j] := Arr[j + 1];
  24.         Arr[j + 1] := TempElement;
  25.       end;
  26.     end;
  27. end;
  28.  
  29. begin
  30.   Randomize;
  31.   for var i := 1 to ArrSize do
  32.     Arr[i] := Random(100);
  33.   Writeln('Исходный массив:', Arr);
  34.  
  35.   BubbleSort(Arr);
  36.   Writeln('Отсортированный массив:', Arr);
  37. end.
