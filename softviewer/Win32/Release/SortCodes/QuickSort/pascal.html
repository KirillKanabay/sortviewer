1. program QuickSortAlgorithm;
  2.  
  3. const
  4.   ArrSize = 10;
  5.  
  6. type
  7.   TArray = array [1..ArrSize] of integer;
  8.  
  9. var
  10.   Arr: TArray;
  11.  
  12. procedure QuickSort(Left, Right: integer);
  13. var
  14.   NewLeft, NewRight : integer; //границы массива
  15.   TempElement, Pivot : integer;
  16. begin
  17.   NewLeft := Left;
  18.   NewRight := Right;
  19.  
  20.   {опорный элемент массива}
  21.   Pivot := Arr[(Left + Right) div 2];
  22.  
  23.   repeat
  24.     while Arr[NewLeft] < Pivot do
  25.       NewLeft := NewLeft + 1;
  26.  
  27.     while Arr[NewRight] > Pivot do
  28.       NewRight := NewRight - 1;
  29.  
  30.     if NewLeft <= NewRight then
  31.     begin
  32.       {обмен значений}
  33.       TempElement := Arr[NewLeft];
  34.       Arr[NewLeft] := Arr[NewRight];
  35.       Arr[NewRight] := TempElement;
  36.  
  37.       NewLeft := NewLeft + 1;
  38.       NewRight := NewRight - 1;
  39.     end;
  40.   until NewLeft > NewRight;
  41.  
  42.   {рекурсивный вызов сортировки для "меньших" элементов}
  43.   if Left < NewRight then
  44.     QuickSort(Left, NewRight);
  45.  
  46.   {сортировка - для "больших" элементов}
  47.   if NewLeft < Right then
  48.     QuickSort(NewLeft, Right);
  49. end;
  50.  
  51. begin
  52.   Randomize;
  53.   for var i := 1 to ArrSize do
  54.     Arr[i] := Random(100);
  55.   Writeln('Исходный массив:', Arr);
  56.  
  57.   QuickSort(1 , ArrSize);
  58.   Writeln('Отсортированный массив:', Arr); 
  59. end.
