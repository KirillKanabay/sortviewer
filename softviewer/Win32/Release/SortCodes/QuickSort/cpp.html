  4. using namespace std;
  5.  
  6. const int arrSize = 10;
  7. int arr[arrSize];
  8.  
  9. void quickSort(int left, int right) {
  10. 	/*Границы массивы*/
  11. 	int newLeft = left;
  12. 	int newRight = right;
  13.  
  14. 	/*Опорный элемент массива*/
  15. 	int pivot = arr[(left + right) / 2];
  16.  
  17. 	do {
  18. 		while (arr[newLeft] < pivot) {
  19. 			newLeft++;
  20. 		}
  21. 		while (arr[newRight] > pivot) {
  22. 			newRight--;
  23. 		}
  24. 		if (newLeft <= newRight) {
  25. 			/*обмен значений*/
  26. 			int tempElement = arr[newLeft];
  27. 			arr[newLeft] = arr[newRight];
  28. 			arr[newRight] = tempElement;
  29.  
  30. 			newLeft = newLeft + 1;
  31. 			newRight = newRight - 1;
  32. 		}
  33. 	} while (newLeft <= newRight);
  34. 	/*рекурсивный вызов сортировки для "меньших" элементов*/
  35. 	if (left < newRight) {
  36. 		quickSort(left, newRight);
  37. 	}
  38. 	/*сортировка - для "больших" элементов*/
  39. 	if (newLeft < right) {
  40. 		quickSort(newLeft, right);
  41. 	}
  42. }
  43.

  51. 	}
  52. 	cout << "Массив до сортировки:";
  53. 	for (int i = 0; i < arrSize; i++) {
  54. 		cout << arr[i] << ' ';
  55. 	}
  56. 	quickSort(0, arrSize - 1);
  57. 	cout << endl;
  58. 	cout << "Массив после сортировки:";
  59. 	for (int i = 0; i < arrSize; i++) {
  60. 		cout << arr[i] << ' ';
  61. 	}
